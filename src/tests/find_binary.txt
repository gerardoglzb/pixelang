program find_binary;

var arr: int[5];
var elementToFind: int;
var index: int;

function binarySearch(element: int, low: int, high: int): int {
    var mid: int;
    while (low <= high) do {
        mid = (low + high) / 2;
        if (arr[mid] == element) {
            return mid;
        } else if (arr[mid] < element) {
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    return -1;
}

{
    arr[0] = 1;
    arr[1] = 2;
    arr[2] = 3;
    arr[3] = 4;
    arr[4] = 5;
    elementToFind = 4;
    index = binarySearch(elementToFind, 0, 4);
    if (index != -1) {
        print("Element", elementToFind, "found at index", index);
    } else {
        print("Element", elementToFind, "not found");
    }
}
