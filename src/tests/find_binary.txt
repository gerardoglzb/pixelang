program find_binary;

var n : int;
var arr: int[5];
var elementToFind: int;
var index: int;

function selectionSort(n : int) : void {
    var i, j, minIndex, temp : int;
    for (i = 0 to n - 1) {
        minIndex = i;
        for (j = i + 1 to n) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j;
            }
        }
        temp = arr[i];
        arr[i] = arr[minIndex];
        arr[minIndex] = temp;
    }
}

function binarySearch(element: int, low: int, high: int): int {
    var mid: int;
    while (low <= high) do {
        mid = (low + high) / 2;
        if (arr[mid] == element) {
            return(mid);
        } else {
            if (arr[mid] < element) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
    }
    return(-1);
}

{
    n = 5;
    arr[0] = 5;
    arr[1] = 3;
    arr[2] = 1;
    arr[3] = 4;
    arr[4] = 2;
    elementToFind = 4;
    selectionSort(n);
    index = binarySearch(elementToFind, 0, n-1);
    if (index != -1) {
        print("Element", elementToFind, "found at index", index);
    } else {
        print("Element", elementToFind, "not found");
    }
}
